def best_shift_signed(base_prof, cur_prof, max_shift):
    w = len(base_prof)
    max_shift = int(min(max_shift, w - 2))
    best_s, best_score = 0, -1e9

    for s in range(-max_shift, max_shift + 1):
        if s >= 0:
            a = base_prof[:w - s]
            b = cur_prof[s:]
        else:
            ss = -s
            a = base_prof[ss:]
            b = cur_prof[:w - ss]

        score = float(np.dot(a, b)) / float(len(a))
        if score > best_score:
            best_score, best_s = score, s

    return best_s, best_score

def aligned_residual(base_prof, cur_prof, shift):
    w = len(base_prof)

        b = cur_prof[shift:]
    else:
        ss = -shift
        a = base_prof[ss:]
        b = cur_prof[:w - ss]
    return float(np.mean(np.abs(a - b)))
