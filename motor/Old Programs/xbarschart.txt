

function main(
  workbook: ExcelScript.Workbook,
  sheetName: string = "Sheet1",
  sampleSize: number = 25,
  skipSize: number = 90,
  imageWidth: number = 800,
  imageHeight: number = 400
): { daysProcessed: number; days: { day: string; xbarImage: string; sImage: string }[] } {

  // ----- Types -----
  type CellValue = string | number | boolean | null;
  type RowValues = CellValue[];
  interface RecordRow {
    date: string | number | null;
    time: string | number | null;
    inputPower: number;
    dateTime: Date;
  }

  // ----- Configuration -----
  const headerRowsToSkip: number = 5;
  const dateCol: number = 3;   // D (zero-based)
  const timeCol: number = 4;   // E
  const powerCol: number = 12; // M

  // X-bar/S constants for n = 25
  const A3: number = 0.606;
  const B3: number = 0.565;
  const B4: number = 1.435;

  // ----- Helpers -----
  const excelDateToJS = (serial: number): Date => {
    const ms = Math.round((serial - 25569) * 86400 * 1000);
    return new Date(ms);
  };

  const parseDateCell = (v: string | number | null): Date | null => {
    if (v === null) return null;
    if (typeof v === "number") return excelDateToJS(v);
    const d = new Date(v);
    return isNaN(d.getTime()) ? null : d;
  };

  const parseTimeText = (v: string | number | null): { h: number; m: number; s: number } | null => {
    if (v === null) return null;
    const parts = String(v).trim().split(":").map(n => parseInt(n, 10));
    if (parts.length < 2 || parts.some(x => isNaN(x))) return null;
    const [h, m, s = 0] = parts;
    return { h, m, s };
  };

  const dayKeyLocal = (d: Date): string => {
    const y = d.getFullYear();
    const m = String(d.getMonth() + 1).padStart(2, "0");
    const dd = String(d.getDate()).padStart(2, "0");
    return `${y}-${m}-${dd}`;
  };

  // ----- Source data -----
  const sheet = workbook.getWorksheet(sheetName);
  if (!sheet) throw new Error(`Worksheet '${sheetName}' not found.`);

  const allValues = sheet.getUsedRange().getValues() as (string | number | boolean)[][];
  const rows: RowValues[] = allValues.slice(headerRowsToSkip);

  // Build typed records
  const records: RecordRow[] = [];
  for (const row of rows) {
    const rawDate = (row[dateCol] ?? null) as (string | number | null);
    const rawTime = (row[timeCol] ?? null) as (string | number | null);
    const ipCell = row[powerCol];

    const ip = typeof ipCell === "number" ? ipCell : parseFloat(String(ipCell));
    if (!Number.isFinite(ip)) continue;

    const d = parseDateCell(rawDate);
    const t = parseTimeText(rawTime);
    if (!d || !t) continue;

    const dt = new Date(d.getFullYear(), d.getMonth(), d.getDate(), t.h, t.m, t.s);
    records.push({ date: rawDate, time: rawTime, inputPower: ip, dateTime: dt });
  }

  // Group by local day
  const groups: Record<string, number[]> = {};
  for (const r of records) {
    const key = dayKeyLocal(r.dateTime);
    if (!groups[key]) groups[key] = [];
    groups[key].push(r.inputPower);
  }

  let dayCount = 0;
  const dayResults: { day: string; xbarImage: string; sImage: string }[] = [];

  // ----- Process each day -----
  for (const [day, powValues] of Object.entries(groups)) {
    const subMeans: number[] = [];
    const subStdDevs: number[] = [];

    // Systematic sampling: take n rows, then skip skipSize rows
    for (let i = 0; i + sampleSize <= powValues.length; i += (sampleSize + skipSize)) {
      const subgroup = powValues.slice(i, i + sampleSize);
      const mean = subgroup.reduce((a, b) => a + b, 0) / sampleSize;
      const variance = subgroup.reduce((acc, v) => acc + (v - mean) ** 2, 0) / (sampleSize - 1);
      subMeans.push(mean);
      subStdDevs.push(Math.sqrt(variance));
    }

    if (!subMeans.length) continue;
    dayCount++;

    const xBarBar = subMeans.reduce((a, b) => a + b, 0) / subMeans.length;
    const sBar    = subStdDevs.reduce((a, b) => a + b, 0) / subStdDevs.length;

    const xUCL = xBarBar + A3 * sBar;
    const xLCL = xBarBar - A3 * sBar;
    const sUCL = B4 * sBar;
    const sLCL = Math.max(0, B3 * sBar);

    // Create/reset the day's sheet
    let daySheet = workbook.getWorksheet(day);
    if (!daySheet) daySheet = workbook.addWorksheet(day);
    daySheet.getUsedRange()?.clear();

    // Write subgroup data
    daySheet.getRange("A1").setValues([["Subgroup", "X-bar", "S"]]);
    const dataRows = subMeans.map((m, idx) => [idx + 1, m, subStdDevs[idx]]);
    daySheet.getRangeByIndexes(1, 0, dataRows.length, 3).setValues(dataRows);

    // Summary limits
    daySheet.getRange("E1").setValues([["X-bar Bar", "UCL X-bar", "LCL X-bar", "S Bar", "UCL S", "LCL S"]]);
    daySheet.getRange("E2").setValues([[xBarBar, xUCL, xLCL, sBar, sUCL, sLCL]]);

    // Categories (A column)
    const cats = daySheet.getRangeByIndexes(1, 0, dataRows.length, 1);

    // --- Limits columns ---
    const clRange  = daySheet.getRangeByIndexes(1, 5, dataRows.length, 1);
    clRange.setValues(Array(dataRows.length).fill([xBarBar]));

    const uclRange = daySheet.getRangeByIndexes(1, 6, dataRows.length, 1);
    uclRange.setValues(Array(dataRows.length).fill([xUCL]));

    const lclRange = daySheet.getRangeByIndexes(1, 7, dataRows.length, 1);
    lclRange.setValues(Array(dataRows.length).fill([xLCL]));

    const sClRange  = daySheet.getRangeByIndexes(1, 8,  dataRows.length, 1);
    sClRange.setValues(Array(dataRows.length).fill([sBar]));

    const sUclRange = daySheet.getRangeByIndexes(1, 9,  dataRows.length, 1);
    sUclRange.setValues(Array(dataRows.length).fill([sUCL]));

    const sLclRange = daySheet.getRangeByIndexes(1, 10, dataRows.length, 1);
    sLclRange.setValues(Array(dataRows.length).fill([sLCL]));

    // ----- X-bar chart (values in B, categories in A) -----
    const xHeaderAndValues = daySheet.getRangeByIndexes(0, 0, dataRows.length + 1, 2); // A1:B(n+1)
    const xChart = daySheet.addChart(ExcelScript.ChartType.lineMarkers, xHeaderAndValues);
    xChart.setPosition("F4");
    xChart.getTitle().setText(`X-bar Chart - ${day}`);
    xChart.getAxes().getCategoryAxis().setTitle("Subgroup");
    xChart.getAxes().getValueAxis().setTitle("X-bar");

    const xCl = xChart.addChartSeries();
    xCl.setName("CL");
    xCl.setXAxisValues(cats);
    xCl.setValues(clRange);

    const xU  = xChart.addChartSeries();
    xU.setName("UCL");
    xU.setXAxisValues(cats);
    xU.setValues(uclRange);

    const xL  = xChart.addChartSeries();
    xL.setName("LCL");
    xL.setXAxisValues(cats);
    xL.setValues(lclRange);

    // ----- S chart (values in C, categories in A) -----
    const sHeaderAndValues = daySheet.getRangeByIndexes(0, 0, dataRows.length + 1, 3); // A1:C(n+1)
    const sChart = daySheet.addChart(ExcelScript.ChartType.lineMarkers, sHeaderAndValues);
    sChart.setPosition("F20");
    sChart.getTitle().setText(`S Chart - ${day}`);
    sChart.getAxes().getCategoryAxis().setTitle("Subgroup");
    sChart.getAxes().getValueAxis().setTitle("S");

    const sClSeries = sChart.addChartSeries();
    sClSeries.setName("CL");
    sClSeries.setXAxisValues(cats);
    sClSeries.setValues(sClRange);

    const sUSeries = sChart.addChartSeries();
    sUSeries.setName("UCL");
    sUSeries.setXAxisValues(cats);
    sUSeries.setValues(sUclRange);

    const sLSeries = sChart.addChartSeries();
    sLSeries.setName("LCL");
    sLSeries.setXAxisValues(cats);
    sLSeries.setValues(sLclRange);

    // ----- Return images (Base64) for this day -----
    const xbarImage: string = xChart.getImage(imageWidth, imageHeight, ExcelScript.ChartImageFittingMode.fit);
    const sImage: string    = sChart.getImage(imageWidth, imageHeight, ExcelScript.ChartImageFittingMode.fit);

    dayResults.push({ day, xbarImage, sImage });
  }  // closes for-loop

  // Final return and closing brace
  return { daysProcessed: dayCount, days: dayResults };
